/* cria arquivo de nomes e endereços organizado 
como uma sequencia (stream) de bytes.
Os campos sao delimitados por | */

#include <stdio.h>
#include <stdlib.h>

#define DELIM_STR "|"

int input (char*, int);

int main() {
    char first[30], last[30], address[30], city[20], state[15], zip[9];
    char filename[15];
    FILE* fd;
	int size;

    printf("Entre com o nome do arquivo a ser criado: ");
    input(filename, 15);

    if ((fd = fopen(filename,"w")) == NULL) {
        printf("Erro na criação do arquivo --- programa abortado\n");
        exit(1);
    }

    printf("\n\nDigite o Sobrenome, ou <ENTER> para sair\n>>>");
    size = input(last, 30);

    while ( size > 0 ) {
        printf("\nPrimeiro nome\n>>>");
        input(first, 30);
        printf("\nEndereco\n>>>");
        input(address, 30);
        printf("\nCidade\n>>>");
        input(city, 20);
        printf("\nEstado\n>>>");
        input(state, 15);
        printf("\nCEP\n>>>");
        input(zip, 9);

        /* envia as strings para o arquivo */
        fputs(last, fd);
        fputs(DELIM_STR, fd);
        fputs(first, fd);
        fputs(DELIM_STR, fd);
        fputs(address, fd);
        fputs(DELIM_STR, fd);
        fputs(city, fd);
        fputs(DELIM_STR, fd);
        fputs(state, fd);
        fputs(DELIM_STR, fd);
        fputs(zip, fd);
        fputs(DELIM_STR, fd);

        /* começa a próxima entrada */
        printf("\n\nDigite o Sobrenome, ou <ENTER> para sair\n>>>");
        size = input(last, 30);
    }

    /* fechar o arquivo */
    fclose(fd);
}

/* função que dá uma linha do stdin para a string str 
*/
int input(char * str, int size) {    
	int i = 0;    
	char c = getchar();    
	while (c != '\n') {        
		if (size > 0) {            
			str[i] = c;            
			i++;            
			size--;        
		}        
		c = getchar();   
	}    
	str[i] = '\0';    
	return i;
}
